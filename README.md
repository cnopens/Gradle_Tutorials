gradle project structrue
 ├─api
 │  └─src

 │      │  ├─java
 │      │  └─resources
 │      └─test
 │          ├─java
 │          └─resources
 ├─dao
 │  └─src
 │      ├─main
 │      │  ├─java
 │      │  └─resources
 │      └─test
 │          ├─java
 │          └─resources
 ├─plugin
 │  └─src
 │      ├─main
 │      │  ├─java
 │      │  └─resources
 │      └─test
 │          ├─java
 │          └─resources
 ├─service
 │  └─src
 │      ├─main
 │      │  ├─java
 │      │  └─resources
 │      └─test
 │          ├─java
 │          └─resources
 ├─shared
 │  └─src
 │      ├─main
 │      │  ├─java
 │      │  └─resources
 │      └─test
 │          ├─java
 │          └─resources
 └─src
     ├─main
     │  ├─java
     │  └─resources
     └─test
         ├─java
         └─resources
~~~
